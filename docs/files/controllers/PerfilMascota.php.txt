    (new NavbarLoggedIn)->render();

        $mascota = (new Mascotas)->getMascota($_GET['id']);
        $turnos = (new Mascotas)->getHistoriaClinica($_GET['id']);
        $v = new PerfilMascota;
        $v->mascota = $mascota;
        $v->turnos = $turnos;
        $v->render();
    } else {
        $v = new PerfilMascotaVacio;
    }
} else {
